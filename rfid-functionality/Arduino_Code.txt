#include <ESP8266HTTPClient.h>
#include <ESP8266WiFi.h>
#include <MFRC522.h>
#include <SPI.h>
#include <WiFiClientSecure.h>
#include <time.h>

#define SS_PIN D2
#define RST_PIN D1

MFRC522 mfrc522(SS_PIN, RST_PIN);
WiFiClientSecure client;

// Replace these values per deployed scanner.
const char* WIFI_SSID = "PrincessChiChi";
const char* WIFI_PASSWORD = "C@ranyagan8891";
const char* HRIS_RFID_ENDPOINT = "https://da-ati.infinityfreeapp.com/hris-system/api/rfid/tap.php";
const char* RFID_DEVICE_CODE = "SCANNER-ATI-HQ-01";
const char* RFID_DEVICE_TOKEN = "ATI-HRIS-RFID-2026-DEVICE-01-9f3d2f6c";

const unsigned long LOCAL_DEBOUNCE_MS = 3000;
String lastUid = "";
unsigned long lastUidSentAt = 0;

String buildCardUid() {
  String uid = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    if (mfrc522.uid.uidByte[i] < 0x10) {
      uid += "0";
    }
    uid += String(mfrc522.uid.uidByte[i], HEX);
  }
  uid.toUpperCase();
  return uid;
}

bool clockIsSynchronized() {
  time_t now = time(nullptr);
  return now > 1700000000;
}

String buildScannedAtIso8601() {
  if (!clockIsSynchronized()) {
    return "";
  }

  time_t now = time(nullptr);
  struct tm* utcTime = gmtime(&now);
  if (utcTime == nullptr) {
    return "";
  }

  char buffer[25];
  strftime(buffer, sizeof(buffer), "%Y-%m-%dT%H:%M:%SZ", utcTime);
  return String(buffer);
}

String jsonEscape(const String& value) {
  String escaped = value;
  escaped.replace("\\", "\\\\");
  escaped.replace("\"", "\\\"");
  return escaped;
}

String extractJsonValue(const String& json, const String& key) {
  String token = "\"" + key + "\":\"";
  int start = json.indexOf(token);
  if (start < 0) {
    return "";
  }

  start += token.length();
  int end = json.indexOf('"', start);
  if (end < 0) {
    return "";
  }

  return json.substring(start, end);
}

void connectWifi() {
  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  Serial.print("Connecting WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(300);
    Serial.print('.');
  }
  Serial.println("\nWiFi connected");
}

void syncClock() {
  configTime(0, 0, "pool.ntp.org", "time.nist.gov");
  Serial.print("Waiting for clock sync");
  unsigned long startedAt = millis();
  while (!clockIsSynchronized() && millis() - startedAt < 10000) {
    delay(250);
    Serial.print('.');
  }
  Serial.println(clockIsSynchronized() ? "\nClock synchronized" : "\nClock sync unavailable, scanner will omit scanned_at");
}

bool postTapToHris(const String& cardUid) {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("WiFi disconnected, reconnecting...");
    connectWifi();
  }

  HTTPClient http;
  client.setInsecure();

  String scannedAt = buildScannedAtIso8601();
  String payload = "{";
  payload += "\"device_code\":\"" + jsonEscape(String(RFID_DEVICE_CODE)) + "\",";
  payload += "\"card_uid\":\"" + jsonEscape(cardUid) + "\"";
  if (scannedAt.length() > 0) {
    payload += ",\"scanned_at\":\"" + jsonEscape(scannedAt) + "\"";
  }
  payload += "}";

  int attempts = 0;
  while (attempts < 3) {
    if (!http.begin(client, HRIS_RFID_ENDPOINT)) {
      Serial.println("Unable to open HRIS endpoint");
      return false;
    }

    http.addHeader("Content-Type", "application/json");
    http.addHeader("X-RFID-DEVICE-TOKEN", RFID_DEVICE_TOKEN);

    int statusCode = http.POST(payload);
    String responseBody = http.getString();
    http.end();

    if (statusCode > 0) {
      String resultCode = extractJsonValue(responseBody, "result_code");
      String action = extractJsonValue(responseBody, "action");
      String message = extractJsonValue(responseBody, "message");

      Serial.print("HTTP status: ");
      Serial.println(statusCode);
      Serial.print("Result code: ");
      Serial.println(resultCode);
      Serial.print("Action: ");
      Serial.println(action);
      Serial.print("Message: ");
      Serial.println(message);

      return statusCode >= 200 && statusCode < 300;
    }

    attempts++;
    Serial.print("Tap POST failed, retry #");
    Serial.println(attempts);
    delay(500);
  }

  return false;
}

void setup() {
  Serial.begin(115200);
  SPI.begin();
  mfrc522.PCD_Init();

  connectWifi();
  syncClock();

  Serial.println("Ready for RFID taps");
}

void loop() {
  if (!mfrc522.PICC_IsNewCardPresent()) {
    return;
  }

  if (!mfrc522.PICC_ReadCardSerial()) {
    return;
  }

  String currentUid = buildCardUid();
  unsigned long now = millis();

  if (currentUid == lastUid && (now - lastUidSentAt) < LOCAL_DEBOUNCE_MS) {
    Serial.println("Duplicate local tap ignored");
    mfrc522.PICC_HaltA();
    mfrc522.PCD_StopCrypto1();
    delay(150);
    return;
  }

  Serial.print("Card UID: ");
  Serial.println(currentUid);

  bool accepted = postTapToHris(currentUid);
  if (accepted) {
    lastUid = currentUid;
    lastUidSentAt = now;
  }

  mfrc522.PICC_HaltA();
  mfrc522.PCD_StopCrypto1();
  delay(300);
}
