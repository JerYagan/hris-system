#include <ESP8266WiFi.h>
#include <ESP8266HTTPClient.h>
#include <SPI.h>
#include <MFRC522.h>
#include <WiFiClient.h>
#include <ArduinoJson.h>

#define SS_PIN D2
#define RST_PIN D1

MFRC522 mfrc522(SS_PIN, RST_PIN);

// Wi-Fi credentials
const char* ssid = "HUAWEI-2.4G-8Wcq";
const char* password = "ytwFW8Jq";

// Direct HRIS RFID API endpoint.
// The ESP8266 must target the LAN IP of the PC running XAMPP, not localhost.
// Update this value if the PC gets a different IP from the router.
String apiUrl = "http://192.168.1.16/hris-system/api/rfid/tap.php";

// Registered device credentials used by the HRIS RFID API.
const char* deviceCode = "SCANNER-ATI-HQ-01";
const char* deviceToken = "ATI-HRIS-RFID-2026-DEVICE-01-9f3d2f6c";

WiFiClient client;

String lastUID = "";

// ------------------------
// HELPER: SEND UID TO HRIS RFID API
// ------------------------
void postTap(String uid) {
  if(uid == "") return;

  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("WiFi disconnected. Tap not sent.");
    return;
  }

  HTTPClient http;

  int attempts = 0;
  while(attempts < 3){
    String payload;
    StaticJsonDocument<192> json;
    json["device_code"] = deviceCode;
    json["card_uid"] = uid;
    serializeJson(json, payload);

    http.setTimeout(5000);
    http.begin(client, apiUrl);
    http.addHeader("Content-Type", "application/json");
    http.addHeader("Accept", "application/json");
    http.addHeader("X-RFID-DEVICE-TOKEN", deviceToken);

    int code = http.POST(payload);
    if(code > 0){
      String response = http.getString();
      Serial.print("API response (HTTP ");
      Serial.print(code);
      Serial.print("): ");
      Serial.println(response);
      http.end();
      break;
    } else {
      Serial.print("Bridge request failed, retrying... ");
      Serial.print(code);
      Serial.print(" (");
      Serial.print(HTTPClient::errorToString(code));
      Serial.println(")");
      http.end();
      attempts++;
      delay(200);
    }
  }

  if (attempts >= 3) {
    Serial.println("RFID API request failed after 3 attempts.");
    Serial.print("Check that Apache is running and reachable at: ");
    Serial.println(apiUrl);
  }
}

// ------------------------
// SETUP
// ------------------------
void setup() {
  Serial.begin(115200);
  SPI.begin();
  mfrc522.PCD_Init();

  WiFi.begin(ssid, password);
  Serial.print("Connecting WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(250);
    Serial.print(".");
  }
  Serial.println("\nWiFi Connected");
  Serial.print("ESP IP Address: ");
  Serial.println(WiFi.localIP());
  Serial.print("RFID API URL: ");
  Serial.println(apiUrl);
  Serial.print("Device Code: ");
  Serial.println(deviceCode);

  Serial.println("Scan RFID Card...");
}

// ------------------------
// LOOP
// ------------------------
void loop() {
  // Check if a new card is present
  if (!mfrc522.PICC_IsNewCardPresent()) {
    lastUID = ""; // reset lastUID when no card present
    return;
  }
  
  if (!mfrc522.PICC_ReadCardSerial()) return;

  // The RFID card/tag provides the UID. The Arduino only reads and forwards it.
  String uid = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    uid += String(mfrc522.uid.uidByte[i], HEX);
  }
  uid.toUpperCase();

  // Only process if new card or card removed before
  if(uid == lastUID) return;

  Serial.print("Card UID: "); Serial.println(uid);

  // Send new tap directly to the HRIS RFID API.
  postTap(uid);

  lastUID = uid;

  mfrc522.PICC_HaltA();
  mfrc522.PCD_StopCrypto1();

  delay(500); // short delay to allow card removal
}
